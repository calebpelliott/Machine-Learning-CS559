Write a short function called my_gradient() that optimizes the weight vector using the squared error function and returns the weight vector when the acceptance of error is below 0.05 and the learning rate $\eta$ is 0.001. [12 pts]

def my_grad(X,Y):
    w1 = 0.0; w2=0.0; eta=0.001
    gradient_w1=0; gradient_w2=0
    for i in range(0,10000):
        y = w1*X**2+w2*X
        error = (Y-y)
        gradient_w1 = np.mean(-X**2*error)
        gradient_w2 = np.mean(-X*error)
        w1 = w1-eta*gradient_w1
        w2 = w2-eta*gradient_w2
        if np.mean(error**2)<=0.05:
            print(w1,w2,np.mean(error**2)/2)
            break
    return w1, w2
